        vector<vector<int>>result;
        vector<int>num;
        unordered_map<int, int> M;
        for (int i = 0; i < nums.size();i++){
            if (M.find(nums[i]) == M.end())
            {
                M[nums[i]] = 1;
                num.push_back(nums[i]);
            }
            else
                M[nums[i]]++;
        }
        sort(num.begin(), num.end());

        {

                    continue;
